127 template<typename T>
128 bool configure_native_thread(T native_handle, ThreadPriority priority, int cpu_id)
129 {
130  bool success = true;
131  sched_param params;
132  int policy;
133  success &= (pthread_getschedparam(native_handle, &policy, &params) == 0);
134  if (priority == ThreadPriority::HIGH) {
135  params.sched_priority = sched_get_priority_max(SCHED_FIFO);
136  } else {
137  params.sched_priority = sched_get_priority_min(SCHED_FIFO);
138  }
139  success &= (pthread_setschedparam(native_handle, SCHED_FIFO, &params) == 0);
140  if (cpu_id >= 0) {
141  cpu_set_t cpuset;
142  CPU_ZERO(&cpuset);
143  CPU_SET(cpu_id, &cpuset);
144  success &= (pthread_setaffinity_np(native_handle, sizeof(cpu_set_t), &cpuset) == 0);
145  }
146 
147  return success;
148 }
